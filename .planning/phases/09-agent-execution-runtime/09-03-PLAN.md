class RunResult(BaseModel):
    status: str  # "completed", "budget_exceeded", "error", "max_tokens_truncated"
    final_text: str
    artifacts: list
    usage: dict
    budget_halt_reason: str | None = None
    error_message: str | None = None
    conversation_history: list[dict]

@runtime_checkable
class RuntimeBackend(Protocol):
    async def execute(
        self, system_prompt: str, messages: list[dict], tools: list[dict],
        tool_dispatch: Callable, budget: BudgetConfig, metering: MeteringCallback,
        agent_id: str, on_turn: Callable | None = None,
    ) -> RunResult: ...

class AgentRunner:
    def __init__(self, backend: RuntimeBackend, router: ToolRouter, output_collector: OutputCollector): ...
    async def run(self, system_prompt: str, initial_prompt: str, budget: BudgetConfig,
                  metering: MeteringCallback, agent_id: str,
                  phase_tools: list[str] | None = None,
                  on_turn: Callable | None = None) -> RunResult: ...

async def dispatch_tool_async(router: ToolRouter, tool_use_block, metering: MeteringCallback) -> dict: ...

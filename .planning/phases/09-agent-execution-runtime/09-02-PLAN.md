      b. Loop:
         ```
         while True:
             response = await self._client.messages.create(
                 model=self._model,
                 max_tokens=self._max_output_tokens,
                 system=system_prompt,
                 tools=tools,
                 messages=messages,
             )

             # Meter
             cost = calculate_cost(response, self._model)
             total_tokens = response.usage.input_tokens + response.usage.output_tokens
             metering.on_llm_call(agent_id_from_context, tokens=total_tokens, cost=cost)
             budget_enforcer.record(tokens=total_tokens, cost=cost)

             # Budget check
             halt_reason = budget_enforcer.is_exceeded()
             if halt_reason:
                 # Preserve partial results
                 break with status="budget_exceeded"

             # Check stop_reason
             if response.stop_reason == "max_tokens":
                 break with status="error", error_message="Response truncated (max_tokens)"

             if response.stop_reason == "end_turn":
                 # Extract final text, break with status="completed"
                 break

             if response.stop_reason == "tool_use":
                 # Extract tool_use blocks
                 tool_use_blocks = [b for b in response.content if b.type == "tool_use"]
                 # Parallel dispatch
                 tool_results = await asyncio.gather(
                     *(tool_dispatch(block) for block in tool_use_blocks)
                 )
                 # Append assistant message + tool results to messages
                 messages.append({"role": "assistant", "content": response.content})
                 messages.append({"role": "user", "content": tool_results})

             # on_turn callback
             if on_turn:
                 await on_turn()
         ```
